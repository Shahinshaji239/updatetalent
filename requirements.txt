# DEPLOYMENT INSTRUCTIONS

# 1. Install Python dependencies:
#    pip install -r requirements.txt

#    cd "react side"
#    npm install

#    cd "react side"

# DATABASE_URL=your-database-url
# CORS_ALLOWED_ORIGINS=your-frontend-url
